    disallowNotOperatorsInConditionals

    Disallows the not, not equals, and strict not equals operators in conditionals.

    Type: Boolean

    Value: true

    Example

    "disallowNotOperatorsInConditionals": true
    Valid
    if (clause) {
    // Do something really crazy
} else {
    // Do something crazy
}

if (a == 1) {
    // Do something really crazy
} else {
    // Do something crazy
}

var a = (clause) ? 1 : 0
    Invalid
    if (!clause) {
    // Do something crazy
} else {
    // Do something really crazy
}

if (a != 1) {
    // Do something crazy
} else {
    // Do something really crazy
}

if (a !== 1) {
    // Do something crazy
} else {
    // Do something really crazy
}

var a = (!clause) ? 0 : 1
