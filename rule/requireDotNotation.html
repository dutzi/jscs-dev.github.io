    requireDotNotation

    Requires member expressions to use dot notation when possible

    Types: Boolean or String

    Values:

    • true
    • "except_snake_case" allow quoted snake cased identifiers

    JSHint: sub

    Example

    "requireDotNotation": true
    Valid for true
    var a = b[c];
var a = b.c;
var a = b[c.d];
var a = b[1];
var a = b['while']; //reserved word
    Valid for "except_snake_case"
    var a = b[c];
var a = b.c;
var a = b['snake_cased'];
var a = b['camelCased_butWithSnakes'];
    Invalid for true
    var a = b['c'];
var a = b['snake_cased'];
var a = b['_camelCased'];
var a = b['camelCased_'];
    Invalid for "except_snake_case"
    var a = b['c'];

    Example for "es3": false or "es3": null

    "requireDotNotation": true,
"es3": false
    Valid
    var a = b[c];
var a = b.c;
var a = b[c.d];
var a = b[1];
var a = b.while;
    Invalid
    var a = b['c'];
var a = b['while']; // reserved words can be property names in ES5
