    disallowNewlineBeforeBlockStatements

    Disallows newline before opening curly brace of all block statements.

    Type: Boolean

    Value: true

    Example

    "disallowNewlineBeforeBlockStatements": true
    Valid
    function good(){
    var obj = {
        val: true
    };

    return {
        data: obj
    };
}

if (cond){
    foo();
}

for (var e in elements){
    bar(e);
}

while (cond){
    foo();
}
