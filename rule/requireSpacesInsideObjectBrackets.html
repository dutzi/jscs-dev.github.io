    requireSpacesInsideObjectBrackets

    Requires space after opening object curly brace and before closing.

    Types: Object or String

    Values: "all" for strict mode, "allButNested" (deprecated use "allExcept": ['}']) ignores closing brackets in a row.

    Example

    "requireSpacesInsideObjectBrackets": {
    "allExcept": [ "}", ")" ]
}

// or
"requireSpacesInsideObjectBrackets": true | "all" | "allButNested"
    Valid for mode "all"
    var x = { a: { b: 1 } };
    Valid for mode "allButNested"
    var x = { a: { b: 1 }};
    Valid for mode "allExcept": [ "}", ")" ]
    var x = { a: (b ? 1 : 2)};
var x = { a: { b: 1 }};
    Invalid
    var x = {a: 1};
