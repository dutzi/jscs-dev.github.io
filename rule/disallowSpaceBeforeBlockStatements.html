    disallowSpaceBeforeBlockStatements

    Disallows space before block statements (for loops, control structures).

    Type: Boolean

    Value: true

    Example

    "disallowSpaceBeforeBlockStatements": true
    Valid
    if (cond){
    foo();
} else{
   bar();
}

for (var e in elements){
    bar(e);
}

while (cond){
    foo();
}
    Invalid
    if (cond) {
    foo();
} else {
   bar();
}

for (var e in elements) {
    bar(e);
}

while (cond) {
    foo();
}
